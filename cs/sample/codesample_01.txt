#各種モジュールをインポート（拡張機能を追加するイメージ）
import scratchattach as scratch3
import time

#セッションIDを決める
sessionID = "ここに取得したセッションIDを入れる"

#Scratchにログイン（正確には少し違います）
session = scratch3.Session(sessionID, username="あなたのユーザー名を入れる")

#クラウド変数に接続（下の数字を作った作品のプロジェクトIDに！）
connection = session.connect_cloud(project_id=907333333)
print("クラウド変数に接続しました")

while True: #無限ループ
  num = str(int(time.time())) #設定する整数を指定（今は1970年からの秒数）
  connection.set_var("last_access", num) #クラウド変数を設定
  print(f"セット完了 {num}")
  time.sleep(5) #5秒待つ
